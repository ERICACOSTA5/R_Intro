5 Aleatorizar tratamientos

5.1 Diseño completamente aleatorizado

library(agricolae)

trt <- c("A", "B", "C")
repeticion <- 4
DCA <- design.crd(trt, r=repeticion, seed=0, serie=0)
(planilha_dic <- DCA$book)

5.2 Diseño en bloques completos aleatorizados

trt <- LETTERS[1:10]
rep <- 5
DCBA <- design.rcbd(trt, r=rep, first=FALSE, seed=111, 
                    kinds = "Super-Duper", serie=2) # write_csv(planilha_DBCA2, "campo.csv")


head(DCBA$book, 10)
print(DCBA$sketch)
